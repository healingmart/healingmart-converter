(() => {
  document.getElementById("hm-converter-self-test-host")?.remove();
  try { delete window.HMConverterSelfTest; } catch (e) { window.HMConverterSelfTest = undefined; }

  [...document.scripts]
    .filter(s => s.src && s.src.includes("hm-converter-self-test"))
    .forEach(s => s.remove());

  s.src =
    "https://healingmart.github.io/healingmart-converter/test/" +
    "hm-converter-self-test.v1.0.2.js?v=" +
    Date.now();
  document.head.appendChild(s);
